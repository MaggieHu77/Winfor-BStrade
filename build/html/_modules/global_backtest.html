[文档]    def complement_single_backtest(self, single_df: pd.DataFrame) -> pd.DataFrame:
        """
        由于个股存在停牌等数据缺失的问题，在进行总表合并的时候，需要先将个股单表按照完整的交易日数据补齐
        :param single_df: 个股回测数据总结表
        :return: 经过日期缺失补足后的个股单表
        """
        new_df = pd.merge(single_df.reset_index(self.INDEX_LEVEL0), self.__empty_pattern,
                          how="outer", left_index=True, right_index=True).sort_index()
        new_df.set_index(pd.MultiIndex.from_arrays([new_df[self.INDEX_LEVEL0], new_df.index],
                                                   names=[self.INDEX_LEVEL0, self.INDEX_LEVEL1]),
                                                   inplace=True)
        specific_values = {"buy_sell": "NA", "operation_price": 0.0, "operation_volume": 0.0}
        # 为防止第一行即为缺失值，全部重新初始化
        new_df.ix[0, ["equity", "cash", "nav", "cumu_r", "volume"]] = [0.0,
                                                                       constant.INIT_FUND,
                                                                       constant.INIT_FUND,
                                                                       0.0,
                                                                       0.0]
        new_df.fillna(value=specific_values, inplace=True)
        new_df.fillna(method="ffill", inplace=True)
        new_df.loc[:, "is_long"] = (new_df.loc[:, "volume"] > 0).astype(int)
        return new_df


    def __run_backtest(self, global_index: str, func: Callable, key_params: Dict[str,
                                                                                      Any]) -> \
            pd.DataFrame:
        """
        回测时间范围内所有成分股票
        :param global_index: 所有待回测股票代码列表
        :param func: 单只股票回测的函数
        :param key_params: func所需参数
        :return: 所有单只股票回测结果合并
        """
        # 获取所有待回测股票代码
        codes = GlobalBacktest.stock_codes_filter(global_index=global_index, date=self.start_time)
        codes_tqdm = tqdm(codes, desc="backtest stocks")
        key_params.update({"begin": self.start_time, "end": self.end_time})
        full_df = pd.DataFrame()
        for cc in codes_tqdm:
            key_params.update({"codename": cc})
            res = func(**key_params)
            if res is None:
                printt(f"## {cc}获取回测结果失败", msg_mode=RunMode.REPORT, global_mode=RUN_MODE)
                continue
            else:
                res = self.complement_single_backtest(single_df=res)
                full_df = full_df.append(res)
        return full_df

    def __calculate_full_account(self, full_df: pd.DataFrame) -> pd.DataFrame:
        """
        计算整个账户的总体信息
        :param full_df: 个股信息拼接成的长表
        :return: 总信息表
        """
        # 加和项
        sum_full_df = full_df.groupby(by=self.INDEX_LEVEL1).sum()
        equity = sum_full_df.equity
        cash = sum_full_df.cash
        nav = sum_full_df.nav
        num_long = sum_full_df.is_long
        position = round(1 - cash / nav, 4)
        # 取平均项
        mean_full_df = full_df.groupby(by=self.INDEX_LEVEL1).mean()
        cumu_r = mean_full_df.cumu_r
        # 记录买卖点
        buy_sell = full_df.groupby(by=self.INDEX_LEVEL1).apply(lambda x: ",".join([t for t in
                                                               x.buy_sell if t != "NA"]))
        full_account_df = pd.DataFrame({"equity": equity,
                                        "cash": cash,
                                        "nav": nav,
                                        "num_long": num_long,
                                        "position": position,
                                        "cumu_r": cumu_r,
                                        "buy_sell": buy_sell})
        return full_account_df

    def __calculate_performance_stats(self, price_seq: pd.Series) -> Dict[str, str]:
        """
        计算价格/价值序列的统计量
        :param price_seq: 价格或价格序列
        :return: 计算结果，包含统计量信息
        """
        rt_seq = price_seq.pct_change()
        annu_rt = rt_seq.mean() * 255
        annu_std = rt_seq.std() * pow(255, 0.5)
        rt_div_std = format(annu_rt / annu_std, ".2%")
        annu_rt = format(annu_rt, ".2%")
        annu_std = format(annu_std, ".2%")
        # 求max_drawdown
        price_array = price_seq.values
        jj = np.argmax(np.maximum.accumulate(price_array) - price_array)
        try:
            ii = np.argmax(price_array[:jj])
        except ValueError:
            ii = 0
        max_drawdown = format(price_array[jj] / price_array[ii] - 1.0, ".2%")
        cumu_rt = format(price_array[-1] / price_array[0] - 1.0, ".2%")
        return {"cumu_return": cumu_rt, "annu_return": annu_rt, "annu_stdev": annu_std,
                "return/stdev": rt_div_std, "max_drawdown": max_drawdown}

    def __get_benchmark_price_seq(self, global_index: str) -> pd.Series:
        """
        获取基准指数回测区间内的日线收盘价数据
        :param global_index: 市场指数代码
        :return:
        """
        # TODO: 替换WinPy接口
        if not w.isconnected():
            w.start()
        res = w.wsd(global_index, "close", self.start_time, self.end_time)
        if res.ErrorCode != 0:
            printt("% `get_benchmark_price_seq`提取数据失败", msg_mode=RunMode.ISSUE, global_mode=RUN_MODE)
            return pd.Series()
        else:
            index = pd.to_datetime(res.Times).date
            price_seq = res.Data[0]
            price_seq = pd.Series(price_seq, index=index)
            return price_seq
