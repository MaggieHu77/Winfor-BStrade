[文档]    def set_paint(self):
        topaint = [False] * len(self.rerun_list)
        if str(PAINT).isdigit():
            if int(PAINT):
                topaint[0: int(PAINT)] = [True] * int(PAINT)
        elif not PAINT:
            pass
        elif ":" in PAINT:
            pp = PAINT.split(":")
            pp = [int(p.strip()) for p in pp]
            assert len(pp) == 2, "输入回测股票图像区间格式错误，期望m:n"
            if int(pp[0]) < len(self.rerun_list):
                topaint[int((pp[0]) - 1): int(pp[1])] = [True] * (int(pp[1]) - int(pp[0]) + 1)
        elif ";" in PAINT:
            pp = PAINT.split(";")
            pp = [p.strip().upper() for p in pp]
            for p in pp:
                if p in self.rerun_list:
                    topaint[self.rerun_list.index(p)] = True
        return topaint



[文档]    def init_stocks(self):
        conn = sqlite3.connect(DATABASE, isolation_level=None, timeout=3)
        cur = conn.cursor()
        cur.execute('''select name from sqlite_master where type="table"''')
        table_list = [ii[0] for ii in cur.fetchall()]
        table_list1 = set([ii.split("_")[1] for ii in table_list])
        # 回测模式下，每次进入都重新更新
        if not MODE:
            for tb in table_list:
                cur.execute(f'''drop table if exists "{tb}"''')
            table_list = []
        if table_list:
            if "Strategy" in table_list:
                table_list.remove("Strategy")
            else:
                create_Strategy((), conn)
            if "Account" in table_list:
                table_list.remove("Account")
            else:
                create_account((), conn)
            self.rerun_list = [ii for ii in self.code_list if ii not in table_list1]
        else:
            create_Strategy((), conn)
            create_account((), conn)
            self.rerun_list = self.code_list
        print("---正在进行数据库初始化\n")
        if self.rerun_list:
            print(f"---检测到{len(self.rerun_list)}需要初始化回测...\n")
            topaint = self.set_paint()
            end_date = [END_DATE, date.today().strftime("%Y-%m-%d")][MODE]
            for code in self.rerun_list:
                try:
                    res = runbacktest(
                        begin=BEGIN_DATE,
                        end=end_date,
                        codename=code,
                        dir=GRAPH,
                        paint=topaint[self.rerun_list.index(code)],
                        star=True
                    )
                except loaddataError as e:
                    print(e)
                    res = None
                if not res:
                    self.code_list.remove(code)
                    continue
                else:
                    trd = res["strategy"][2]
                    if not trd:
                        print(f"---Waring:{code}趋势信息缺失，请检查回测结果；"
                              f"继续运行策略将剔除该股票")
                        self.code_list.remove(code)
                        continue
                create_stock_hl(res["hlp"], conn, code)
                create_stock_bs(res["bsp"], conn, code)
                cur.execute(f'''delete from Strategy where code="{code}"''')
                cur.execute('''insert into  Strategy values(?,?,?,?,?,?,?,?,?,?)''', res["strategy"])
                cur.execute(f'''delete from Account where code="{code}"''')
                cur.execute('''insert into Account values(?,?,?,?,?,?)''', res["account"])
        print("---数据初始化完成")
        cur.close()
        conn.close()

[文档]    def job_30min(self):
        conn = sqlite3.connect(DATABASE, isolation_level=None, timeout=3)
        cur = conn.cursor()
        cur.execute(f'''select code from Strategy where operate_lev > 1''')
        list_30min = [ii[0] for ii in cur.fetchall()]
        if list_30min:
            print(f"30min job:{len(list_30min)}只股票待更新")
            stamp = datetime.now()
            if not w.isconnected():
                w.start()
            begin_time = (stamp - timedelta(seconds=300)).strftime("%Y-%m-%d %H:%M:%S")
            end_time = (stamp + timedelta(minutes=1)).strftime("%Y-%m-%d %H:%M:%S")
            highs = w.wsi(",".join(list_30min), "high", begin_time, end_time, "barsize=30")
            tick = highs.Times[0].strftime("%Y-%m-%d %H:%M:%S")
            highs = highs.Data[0]
            lows = w.wsi(",".join(list_30min), "low", begin_time, end_time, "barsize=30").Data[0]
            for code in list_30min:
                print(f"30min job:正在更新{code}")
                tt_i = list_30min.index(code)
                high = highs[tt_i]
                low = lows[tt_i]
                if not high or not low or high == nan or low == nan:
                    print(f"30min job:更新中出现无效最高价和最低价数据，high:{high}, low:{low}")
                    continue
                # 高低点部分
                hlp = self.wrap_hlp(code, conn)
                hlp["30min"].update({"high": high, "low": low, "price": round((high + low) / 2, 2)})
                hl_res = HLPoint.step_hl_s(**hlp["30min"], stamp=tick)
                if hl_res["is_high"]:
                    cur.execute(f'''insert into 'HL_{code}' values(?,?,?,?,?)''',
                                (hlp["30min"]["temp_hl_t"], hlp["30min"]["temp_hl"], "L", "30min", tick))
                    hlp["30min"].update({hl_res})
                elif hl_res["is_low"]:
                    cur.execute(f'''insert into 'HL_{code}' values(?,?,?,?,?)''',
                                (hlp["30min"]["temp_hl_t"], hlp["30min"]["temp_hl"], "H", "30min", tick))
                    hlp["30min"].update({hl_res})
                else:
                    hlp["30min"].update({hl_res})
                # 买卖点部分
                bsp = self.wrap_trade(code, conn)
                next_i = bsp["last_bs"][1] + 1
                account1 = self.wrap_account(code, conn)
                account = Trade.Account(volume=account1["volume"],
                                        equity=account1["equity"],
                                        cumu_r=account1["cumu_r"])
                if bsp["operate_lev_sell"] >= 2 and low < hlp["30min"]["pre_low"]:
                    for bb in bsp["order_l"]:
                        if bb[8] >= 2:
                            account1["cash"] += hlp["30min"]["price"] * bb[3]
                            account1["volume"] -= bb[3]
                            account1["equity"] = account1["volume"] * hlp["30min"]["price"]
                            account1["cumu_r"] = (account1["cash"] + account1["equity"]) / INIT_FUND - 1

                            cur.execute(f'''insert into 'BS_{code}' values(?,?,?,?,?,?,?,?,?,?)''',
                                        (tick, next_i, hlp["30min"]["price"],
                                         bb[3], "S4", 0.0, 0.0, 0.0, -1, bb[1]))
                            cur.execute(f'''update 'BS_{code}' set counter_opt={next_i} where index={bb[1]}''')
                if bsp["operate_lev_buy"] == 2:
                    res_buy = Trade.seek_to_buy_s(pre_low=hlp["daily"]["pre_low"], pre_high30=hlp["30min"]["pre_high"],
                                        pre_low30=hlp["30min"]["pre_low"], num_h30=hlp["30min"]["num_h30"],
                                        num_l30=hlp["30min"]["num_l30"], stamp=tick, trend=bsp["trend"],
                                        pre_trend=bsp["pre_trend"], index=next_i, account=account,
                                        high=high, low=low, price=hlp["30min"]["price"],
                                        last_bs_typ=bsp["last_bs"][4])
                    if res_buy[0]:
                        pass



        cur.close()
        conn.close()
        pass

[文档]    def wrap_hlp(self, code, conn):
        cur = conn.cursor()
        cur.execute(f'''select * from Strategy where code="{code}"''')
        row = dict(zip(Col_stg, cur.fethone()))
        hlp = {"daily": {}, "30min": {}, "5min": {}}
        hlp["daily"].update(dict(zip([cc.rstrip("d") for cc in Col_stg_hlp_d],
                                     [row[cc] for cc in Col_stg_hlp_d])))
        hlp["30min"].update((dict(zip([cc.rstrip("30") for cc in Col_stg_hlp_30],
                                      [row[cc] for cc in Col_stg_hlp_30]))))
        hlp["5min"].update((dict(zip([cc.rstrip("5") for cc in Col_stg_hlp_5],
                                     [row[cc] for cc in Col_stg_hlp_5]))))
        cur.execute(f'''select num_h30, num_l30, num_h5, num_l5 from Strategy where code="{code}"''')
        num = cur.fetchone()
        hlp["30min"].update({"num_h30": num[0], "num_l30": num[1]})
        hlp["5min"].update({"num_h5": num[2], "num_l5": num[3]})
        cur.close()
        return hlp

[文档]    def wrap_trade(self, code, conn):
        cur = conn.cursor()
        bsp = {}
        cur.execute(f'''select * from "BS_{code}" order by date desc''')
        bsp.update({"last_bs": cur.fethone()})
        cur.execute(f'''select * from "BS_{code}" where type like "B%" and counter_opt>0''')
        bsp.update({"order_l": cur.fetchall()})
        cur.execute(f'''select operate_lev, operate_lev_buy, operate_lev_sell, trend, pre_trend from Strategy 
        where code="{code}"''')
        bsp.update(dict(zip(["operate_lev", "operate_lev_buy", "operate_lev_sell", "trend", "pre_trend"], cur.fetchone())))
        cur.close()
        return bsp
